                for(i = 0; i < array.length; i++)
                {
                for(j = 0; j < array.length; j++)

                break;
                }
                if(j == array.length )
                {
                ++count;

                }

function Birthday_Game(arr,D ,M) {
    var num = 0;
             for(let i = 0;i< arr.length - M+1;i++){
             let slicedSquare = arr.slice(i,i + M);
               let sum = slicedSquare.reduce((total,number) => total+number,0);
               
               if(sum === D){
                 num++;
               }
             }
             
             return num;
  
  }
